-- Create SAILORS table
CREATE TABLE SAILORS (
    sid INT PRIMARY KEY,
    name VARCHAR(50),
    rating INT,
    age INT
);

-- Create BOATS table
CREATE TABLE BOATS (
    bid INT PRIMARY KEY,
    bname VARCHAR(50),
    color VARCHAR(20)
);

-- Create RESERVES table
CREATE TABLE RESERVES (
    sid INT,
    bid INT,
    day VARCHAR(10),
    PRIMARY KEY (sid, bid),
    FOREIGN KEY (sid) REFERENCES SAILORS(sid),
    FOREIGN KEY (bid) REFERENCES BOATS(bid)
);

-- Create Teacher table
CREATE TABLE Teacher (
    Tid INT PRIMARY KEY,
    Name VARCHAR(50),
    Dept VARCHAR(50)
);

-- Create Subject table
CREATE TABLE Subject (
    Subno INT PRIMARY KEY,
    Subtitle VARCHAR(50)
);

-- Create Taught_By table
CREATE TABLE Taught_By (
    Lid INT,
    Subno INT,
    PRIMARY KEY (Lid, Subno),
    FOREIGN KEY (Lid) REFERENCES Teacher(Tid),
    FOREIGN KEY (Subno) REFERENCES Subject(Subno)
);

-- Create Student table
CREATE TABLE Student (
    Rollno INT PRIMARY KEY,
    Sname VARCHAR(50),
    City VARCHAR(50)
);

SELECT b.color
FROM BOATS b
JOIN RESERVES r ON b.bid = r.bid
JOIN SAILORS s ON r.sid = s.sid
WHERE s.name = 'Tarun';

SELECT s.sid, s.name
FROM SAILORS s
JOIN RESERVES r ON s.sid = r.sid
WHERE r.day = 'Monday';

SELECT bid, bname
FROM BOATS
WHERE color IN ('Red', 'Green');

DELETE FROM SAILORS
WHERE age > 60;





SELECT T.Name
FROM Teacher T
JOIN Taught_By TB ON T.Tid = TB.Lid
JOIN Subject S ON TB.Subno = S.Subno
WHERE T.Dept = 'Physics' AND S.Subtitle = 'Thermodynamics';
